using namespace std;
ifstream fin("inima.in");

const int N = 100000;
int p[N], n;
int main()
{
fin >> n;
for(int i = 0; i < n; ++i)
fin >> p[i];
int act = 0, w, h;
int p1, p2;
for (int i = 0, j = n - 1; i < j; )
{
w = j - i;
h = min(p[i], p[j]);
act = max(act, w * h);
p1 = p[i], p2 = p[j];
if (p1 <= p2)
{
while (i < j && p[i] <= p1)
++i;
if (i < j) continue;
}
